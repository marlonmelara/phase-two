Entendiendo las Variables de CSS

/* --- ÁMBITO GLOBAL --- */
:root {
  --brand-color: #00e5ff; /* Celeste brillante global */
  --text-primary: #ffffff; /* Texto blanco global */
  --padding-base: 1rem; /* Escala de espaciado reutilizable */
}

/* --- ESTILOS COMPONENTES --- */
.btn-primary {
  background-color: var(--brand-color, #ff007f);
  color: var(--text-primary, #000000);
  padding: var(--padding-base);
  border: none;
  border-radius: 4px;
}

/* --- ÁMBITO LOCAL --- */
.alert-box {
  --brand-color: #ff3d00; /* Sobreescritura local a Rojo Alerta */
  --padding-base: 1.5rem; /* El padding aumenta solo para este bloque */

  background-color: #1e1e1e;
  padding: var(--padding-base); /* Aplica las 1.5rem locales */
  border-left: 5px solid var(--brand-color);
}
